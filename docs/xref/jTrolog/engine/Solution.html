20  package jTrolog.engine;
21  
22  import jTrolog.terms.Term;
23  
24  import java.util.HashMap;
25  import java.util.Iterator;
26  
27  /**
28   * @author ivar.orstavik@hist.no
29   */
30  @SuppressWarnings({ "rawtypes" })
31  public class Solution {
32  
33  	private static final HashMap emptyMap = new HashMap();
34  
35  	HashMap bindings;
36  	Term solution;
37  
38  	public Solution(Term solution) {
39  		this.bindings = emptyMap;
40  		this.solution = solution;
41  	}
42  
43  	public Solution(HashMap bindings, Term solution) {
44  		this.bindings = bindings;
45  		this.solution = solution;
46  	}
47  
48  	/**
49  	 * @return true if the query was a success, false otherwise
50  	 */
51  	public boolean success() {
52  		return solution != null;
53  	}
54  
55  	/**
56  	 * @return the solution to the query as a Term
57  	 */
58  	public Term getSolution() {
59  		return solution;
60  	}
61  
62  	/**
63  	 * @return the link of the Variable corresponding to varName, if no Var was
64  	 *         named varName in the query, null is returned if the Var was
65  	 *         linked to an any Var, that any Var is returned
66  	 */
67  	public Term getBinding(String varName) {
68  		return (Term) bindings.get(varName);
69  	}
70  
71  	public String toString() {
72  		return solution == null ? "no" : solution.toString();
73  	}
74  
75  	public String bindingsToString() {
76  		StringBuffer buffy = new StringBuffer();
77  		for (Iterator it = bindings.keySet().iterator(); it.hasNext();) {
78  			String variable = (String) it.next();
79  			Term binding = (Term) bindings.get(variable);
80  			buffy.append(variable).append(": ").append(binding).append("\n");
81  		}
82  		return buffy.toString();
83  	}
84  }
